// server/src/services/cache.service.ts

const CACHE_TTL = {
  PACK_LIST: 300,        // 5 minutes
  PACK_DETAIL: 600,      // 10 minutes  
  CATEGORY_LIST: 3600,   // 1 hour
  FEATURED_PACKS: 900,   // 15 minutes
  TRENDING: 300,         // 5 minutes
  USER_PURCHASES: 60,    // 1 minute (more dynamic)
}

export class CacheService {
  async get<T>(key: string): Promise<T | null>
  async set(key: string, value: unknown, ttl: number): Promise<void>
  async del(key: string): Promise<void>
  async delPattern(pattern: string): Promise<void>
  
  // Cache-aside wrapper
  async remember<T>(key: string, ttl: number, fn: () => Promise<T>): Promise<T> {
    const cached = await this.get<T>(key)
    if (cached !== null) return cached
    const fresh = await fn()
    await this.set(key, fresh, ttl)
    return fresh
  }
}

── F.2 Pre-Deploy Script ──

□ Create scripts/pre-deploy-check.ts:

  async function preDeployCheck() {
    const errors: string[] = []
    const warnings: string[] = []
    
    // Required env vars
    const required = ['DATABASE_URL', 'JWT_SECRET', 'JWT_REFRESH_SECRET', 'NODE_ENV']
    required.forEach(key => {
      if (!process.env[key]) errors.push(`Missing required: ${key}`)
    })
    
    // JWT secret strength
    if ((process.env.JWT_SECRET?.length ?? 0) < 32) {
      errors.push('JWT_SECRET must be at least 32 characters')
    }
    
    // Production checks
    if (process.env.NODE_ENV === 'production') {
      if (!process.env.SENTRY_DSN) warnings.push('SENTRY_DSN not set — errors wont be tracked')
      if (process.env.STRIPE_SECRET_KEY?.startsWith('sk_test_')) {
        warnings.push('Using Stripe TEST key in production')
      }
      if (process.env.CORS_ORIGIN === '*') {
        errors.push('CORS_ORIGIN cannot be wildcard in production')
      }
    }
    
    // Database connectivity
    try {
      await prisma.$queryRaw`SELECT 1`
      console.log('✅ Database connection: OK')
    } catch (e) {
      errors.push(`Database connection failed: ${e.message}`)
    }
    
    // Redis connectivity
    try {
      await redis.ping()
      console.log('✅ Redis connection: OK')
    } catch (e) {
      warnings.push(`Redis connection failed: ${e.message}`)
    }
    
    if (errors.length > 0) {
      console.error('❌ Pre-deploy checks FAILED:')
      errors.forEach(e => console.error(`  - ${e}`))
      process.exit(1)
    }
    
    if (warnings.length > 0) {
      console.warn('⚠️  Warnings:')
      warnings.forEach(w => console.warn(`  - ${w}`))
    }
    
    console.log('✅ All pre-deploy checks passed')
    process.exit(0)
  }
